    # Vue 2.x

    # 1. v-html 和 domPropsInnerHTML

    • template写法：要直接渲染一段html，是使用vue的v-html指令
    • jsx写法：与之对应的是domPropsInnerHTML
    <template>
    <div v-html="remark">
</template>

<script>
export default {
    data () {
      return {
        remark: `<span style="color: red">测试html</span>`
      }
    },
    render () {
      return <p domPropsInnerHTML={this.remark} />
    }
}
</script>

    # 2. 绑定变量，事件

    • template写法：绑定变量为(v-bind):data="someData"，绑定事件为v-on:change="onChange"或简写@change="onChange"
    • jsx写法：绑定变量为data={this.someData}，绑定事件为onChange={this.onChange}
    <template>
  <div :data="someData" @change="onChange">
</template>

<script>
export default {
  data () {
    return {
      someData: 'test'
    }
  },
  render () {
    return <div data={this.someData} onChange={this.onChange} />
  },
  methods: {
    onChange (e) {
      console.log(e)
    }
  }
}
</script>

    # 3. v-for 和 map

    • template写法：渲染多个同样表现的元素，一般采用v-for完成，但不提倡v-for与v-if写在同一级上，同时如果是直接在template上v-for，循环元素时的key应该写在template下第一个非template的元素。
    • jsx写法：比v-for要灵活一些，直接arr.map(item => <p>{item}</p>)即可，利用的是数组的map方法
    <template>
  <ul>
    <li v-for="item in arr" :key="item.id">
      {{ item.name }}
    </li>
  </ul>
</template>

<script>
export default {
  data() {
    return {
      arr: [
        { id: 1, name: "1" },
        { id: 2, name: "2" },
        { id: 3, name: "3" },
      ],
    };
  },
  render() {
    return (
      <ul>
        {this.arr.map((item) => (
          <li>{item.name}</li>
        ))}
      </ul>
    );
  },
};
</script>

    # 4. v-if 和三元表达式

    • template写法：直接在对应的元素上写v-if="条件"即可，通过v-if控制的元素不符合条件直接不渲染，而不是v-show的display: none
    • jsx写法：使用三元表达式，符合条件返回元素，不符合条件返回null
    <template>
  <div v-if="flag">我出现了</div>
</template>

<script>
export default {
  data() {
    return {
      flag: true,
    };
  },
  render() {
    return this.flag ? <div>我出现了</div> : null;
  },
};
</script>

    # 5. v-show 和 style: display

    • template写法：直接v-show后面绑定条件变量或表达式即可，控制css的display是否为none
    • jsx写法：显示的话不需设置display，不显示的话display: none即可达到一样效果

    # 6. v-slot 和 scopedSlots

    • template写法：一般为了不影响样式直接在<template>上写slot名字，然后slot-scope="变量",就能取到组件内部绑定到插槽上的值了，更推荐是直接写v-slot:#插槽名
    • jsx写法：直接在对应的组件上写上scopedSlots，绑定一个对象，每个对象返回一段html即可，同时，不具名插槽的key为default
    <template>
  <a-dropdown placement="bottomCenter">
    <template slot="overlay">
      <a-menu>
        <a-menu-item>
          <a-button type="link" icon="form" size="small" @click="createBatch">
            批量创建
          </a-button>
        </a-menu-item>
        <a-menu-item>
          <a-button
            type="link"
            icon="delete"
            @click="deleteBatch"
            size="small"
            v-if="selectedRowKeys.length"
          >
            批量删除
          </a-button>
        </a-menu-item>
      </a-menu>
    </template>
    <a-button icon="down" type="primary"> 批量操作 </a-button>
  </a-dropdown>
</template>

<script>
export default {
  data() {
    return {
      selectedRowKeys: [],
    };
  },
  render() {
    return (
      <a-dropdown
        placement="bottomCenter"
        scopedSlots={{
          overlay: (props) => {
            return (
              <a-menu>
                <a-menu-item>
                  <a-button
                    type="link"
                    icon="form"
                    size="small"
                    onClick={this.createBatch}
                  >
                    批量创建
                  </a-button>
                </a-menu-item>
                {this.selectedRowKeys.length > 0 ? (
                  <a-menu-item>
                    <a-button
                      type="link"
                      icon="delete"
                      onClick={this.deleteBatch}
                      size="small"
                    >
                      批量删除
                    </a-button>
                  </a-menu-item>
                ) : null}
              </a-menu>
            );
          },
        }}
      >
        <a-button icon="down" type="primary">
          批量操作
        </a-button>
      </a-dropdown>
    );
  },
  methods: {
    createBatch(e) {
      console.log(e);
    },
    deleteBatch(e) {
      console.log(e);
    },
  },
};
</script>

    # 7. v-bind 和...{ props }

    • template写法：除了单个简写冒号的绑定prop外，还可以一次性v-bind绑定一个对象，对象上的key需对应组件props里的key，否则就会落入$attrs即标签的attributes中
    • jsx写法：要实现上述写法，需要借助es6的...扩展运算符
    <template>
  <a-modal v-bind.sync="modal" @cancel="cancelDialog">
    <div>测试</div>
  </a-modal>
</template>

<script>
export default {
  data() {
    return {
      modal: {
        confirmLoading: false,
        destroyOnClose: true,
        keyboard: false,
        maskClosable: false,
        dialogClass: "mydialog",
        title: "",
        visible: false,
        wrapClassName: "mywrapper",
        centered: true,
        footer: null,
        components: "",
        width: "50%",
      },
    };
  },
  render() {
    return (
      <a-modal {...{ props: this.modal }} onCancel={this.cancelDialog}>
        <div>测试</div>
      </a-modal>
    );
  },
  methods: {
    cancelDialog(e) {
      console.log(e);
    },
  },
};
</script>

    # 8. v-model 和 value + 事件

    • template写法：v-model支持表单元素，监听组件的input或change事件，自定义组件加上model后也可以支持v-model
    • jsx写法：v-model是vue指令，是一个语法糖，jsx实现该效果，则是传值进prop，然后在响应事件里接收弹出来的值即可
    <template>
  <a-select v-model="value" v-bind="{ options, showSearch }">
</template>

<script>
export default {
  data () {
    return {
      value: ''
      options: [],
      showSearch: true
    }
  },
  render () {
    const props = {
      options,
      showSearch
    }
    return <a-select value={this.value} {...{props}} onChange={console.log}/>
  }
}
</script>

    # 9. prop.sync 和 on=update:prop

    • template写法：prop.sync是vue的一个支持组件内同步改值的修饰符，正常情况下，子组件除了可以直接改对象的某一key: value，其他均需通知父组件来修改，使用.sync可以解决
    • jsx写法：jsx绑定变量和事件是没有修饰符这些的，但.sync是响应update:变量名这个事件，所以可以从这一步出发来解决问题
    <template>
  <MyComponent myvalue.sync="myvalue" />
</template>

<script>
import MyComponent from "xxx";
export default {
  components: {
    MyComponent,
  },
  data() {
    return {
      myvalue: "",
    };
  },
  render() {
    return (
      <MyComponent
        myvalue={this.myvalue}
        on={{
          "update:myvalue": console.log,
        }}
      />
    );
  },
};
</script>

    # 10. 事件是否带括号

    • template写法：事件是否带括号还要看下方的函数是怎么设置接参的，如果是与组件弹出来的个数一致，不用带括号也可取到，如果要带多个，可使用箭头函数处理即可；单独带括号不传变量，仅是函数里取不到值
    • jsx写法：jsx绑定事件，不能带括号，如需带多参数，均需使用箭头函数来处理，直接方法括号，会渲染时就被调用从而报错

    # Vue 3

    # 1. v-bind 和 ...{}

    • template写法：vue3.0层级props的抹平，对v-bind没有影响，与vue2.x写法一致，如需就地改值，无需使用.sync，直接v-model即可
    • jsx写法：vue2.0是需要{... {props}}才能达到与v-bind一样的效果，vue3.0直接{... { A: a, B: b }}即可，没有层级
    <template>
  <el-table v-bind="tableProps">
    <el-table-column v-for="(column, index) in columns" :key="index">
      <template v-if="$slots[column.header]" v-slot:header="props">
        <slot :name="column.header" v-bind="props" />
      </template>
      <template v-if="$slots[column.content]" v-slot="props">
        <slot :name="column.content" v-bind="props">
      </template>
    </el-table-column>
  </el-table>
</template>

<script>
import ElTable from 'element-plus/lib/el-table'
import { computed, defineComponent } from "vue";
import _ from 'lodash'
export default defineComponent({
  name: 'test'
  props: Object.assign({}, ElTable.props, {
    columns: {
      type: Array,
      default: () => []
    }
  }),
  setup (props) {
    const tableProps = computed(() =>
            Object.assign({},
                _.pick(props, Object.keys(ElTable.props)),
                ElTable.emits?.reduce((self: any, item) => {
                    const key = `on${item.split('-').map(str => str[0].toUpperCase() + str.slice(1)).join('')}`
                    self[key] = function () {
                        return emit(item, arguments)
                    }
                    return self
                }, {})
            )
        )
      return {
        tableProps
      }
  },
  render () {
    const slots = this.$slots as any
    return <ElTable {...this.tableProps}>
                {this.columns.map((column: any, index: number) => <el-table-column {...column} key={index} >
                    {Object.assign({}, slots[column.header] && {
                        header: (props: any) => slots[column.header](props)
                    }, slots[column.content] && {
                        default: (props: any) => slots[column.content](props)
                    })}
                </el-table-column>)
                }
                {slots.append && {
                    append: () => slots.append(this.data)
                }}
            </ElTable>
  }
})
</script>

    # 2. v-model

    • template写法：vue3.0中，v-model可以连续绑定多个变量，如v-model:a，v-model:b，不写的话v-model等价于v-model:modelValue，如果是自定义组件，要实现v-model，需要在自定义组件更新处写上emit('update:xxx', 值)即可
    • jsx写法：大体同上，但jsx不能处理v-model:xxx，如需v-model:xxx，则要么
      • ① 组件上xxx={变量}，然后再响应{...{ onUpdate:xxx: (新值) => 变量 = 新值}}，这种较为麻烦
      • ② v-model={[this.visible, 'visible']}既能实现v-model:visible="visible"的效果，这种的话很方便，但比较难记住
    <template>
  <RightContextMenu
    v-model:visible="visible"
    :top="top"
    :left="left"
    :menus="menus"
  />
</template>

<script>
import RightContextMenu from "./RightContextMenu";
import { defineComponent, ref } from "vue";
export default defineComponent({
  name: "test",
  components: {
    RightContextMenu,
  },
  setup() {
    const visible = ref < any > false;
    const top = ref < any > 0;
    const left = ref < any > 0;
    const menus = [];
    return {
      visible,
      top,
      left,
      menus,
    };
  },
  render() {
    return (
      <RightContextMenu
        v-model={[this.visible, "visible"]}
        top={this.top}
        left={this.left}
        menus={this.menus}
      />
    );
  },
});
</script>
