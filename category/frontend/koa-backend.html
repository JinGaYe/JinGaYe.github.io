    # 1. 初始化项目

    mkdir yourProject
cd yourProject
npm init -y

    # 2. 安装必要的 npm 模块

    npm install -s fs koa mysql2
npm install -s uuid nodemailer nunjucks
npm install -s js-sha256 koa-router
npm install -s jsonwebtoken koa-session koa-websocket
npm install -s koa2-cors koa2-swagger-ui sequelize
npm install -s svg-captcha koa-bodyparser swagger-jsdoc

    说明

    1. fs：读取文件使用
    2. koa: 框架
    3. mysql2：操作数据库需要用到的依赖
    4. uuid：生成唯一标识使用
    5. nodemailer：使用 node 发送邮件使用的依赖
    6. nunjucks：挂载静态模板
    7. js-sha256：加密
    8. koa-router：管理接口路径
    9. jsonwebtoken：生成 token 使用
    10. koa-session：用于 session 存放信息校验
    11. koa-websocket：封装 websocket 的模块
    12. koa2-cors：处理跨域请求
    13. koa2-swagger-ui：绘制 swagger 文档
    14. sequelize：操作数据库的 orm 模块
    15. svg-captcha：生成 svg 验证码
    16. koa-bodyparser：处理 post 请求的内容
    17. swagger-jsdoc：生成 swagger 文档必要的模块
    • 其中，swagger-jsdoc安装最新版本可能导致node无法识别import导致项目启动不了，鉴于尝试多种babel的处理方式都不得成功，故采用折衷方法，降级到6.0.2版本。

    # 4. 封装统一管理 app.js

    const koa = require("koa");
const bodyParser = require("koa-bodyparser");
const cors = require("koa2-cors");
const vo = require("./utils/vo");
const { configSession } = require("./utils/session");
const WebSocket = require("koa-websocket");
const ChatApi = require("./services/ws");
const { koaSwagger } = require("koa2-swagger-ui");
const check = require("./utils/check");
const app = WebSocket(new koa());

app.use(
  koaSwagger({
    routePrefix: "/", // host at /swagger instead of default /docs
    swaggerOptions: {
      url: "/swagger.json", // example path to json 其实就是之后swagger-jsdoc生成的文档地址
    },
  })
);
configSession(app);
app.use(bodyParser());
app.use(
  cors({
    credentials: true, //默认情况下，Cookie不包括在CORS请求之中。设为true，即表示服务器许可Cookie可以包含在请求中
    origin: (ctx) => ctx.header.origin, // web前端服务器地址，注意这里不能用*
  })
);
app.use(vo());
app.use(check);

const controller = require("./controller")();
app.use(controller.routes(), controller.allowedMethods());
ChatApi(app);

app.listen(3000);

    # 5. 封装处理 swagger，router

    const fs = require("fs");
const swaggerJSDoc = require("swagger-jsdoc");
// 定义swagger的一些配置
const swaggerDefinition = {
  info: {
    title: "API",
    version: "1.0.0",
    description: "Vuepress-Sakura-blog使用的后台api接口",
  },
  host: "localhost:3000",
};

// 初始化swagger，把整理好的router接口塞进来
const initSwagger = (router) => {
  const options = {
    swaggerDefinition,
    apis: ["./controllers/*.js"],
  };
  const swaggerSpec = swaggerJSDoc(options);
  router.get("/swagger.json", async (ctx) => {
    ctx.set("Content-Type", "application/json");
    ctx.body = swaggerSpec;
  });
};

// 处理好接口的请求路径
const addMapping = (router, mapping) => {
  for (const url in mapping) {
    if (url.startsWith("GET ")) {
      const path = url.substring(4);
      router.get(path, mapping[url]);
    } else if (url.startsWith("POST ")) {
      const path = url.substring(5);
      router.post(path, mapping[url]);
    } else {
    }
  }
};

// 批量处理多个接口文件
const addController = (router) => {
  const files = fs.readdirSync(__dirname + "/controllers");
  const jsfiles = files.filter((f) => f.endsWith(".js"));
  for (const f of jsfiles) {
    addMapping(router, require(__dirname + "/controllers/" + f));
  }
};

// 返回router对象
module.exports = (dir = "controllers") => {
  const router = require("koa-router")();
  addController(router, dir);
  initSwagger(router);
  return router;
};

    # 6. 封装接口标准返回 vo

    const routerResponse = (options = {}) => {
  return async (ctx, next) => {
    ctx.success = (data, message = "success") => {
      ctx.type = options.type || "json";
      ctx.body = {
        code: options.code || 200,
        message,
        success: true,
        result: data,
      };
    };

    ctx.fail = (message = "fail", code = 500) => {
      ctx.type = options.type || "json";
      ctx.body = {
        code,
        message,
        success: false,
        result: null,
      };
    };

    await next();
  };
};

module.exports = routerResponse;

    # 7. 搭建简易websocket

    const { addChat } = require('./chat')
const queue = {}
const chat = app => {
    app.ws.use((ctx, next) => {
        ctx.websocket.on('message', async msg => {
            const result = await addChat(JSON.parse(msg))
            if (!queue[result.uid]) {
                ctx._uid = result.uid
                queue[result.uid] = ctx
            }
            Object.values(queue).forEach(({ websocket }) => {
                if (result !== null) {
                    websocket.send(JSON.stringify(['uid', 'username', 'content', 'createDate', 'avatar', 'style'].reduce((self, key) => {
                        self[key] = result[key]
                        return self
                    }, {})))
                }
            })
        })
        ctx.websocket.on('close', async () => {
            if (queue[ctx._uid]) {
                delete queue[ctx._uid]
            }
        })
    })
}

module.exports = chat

    # 9.部分接口的编写chat.js

    const { queryAllUser } = require('../services/user')
const { queryAllChat } = require('../services/chat')

/**
 * @swagger
 *
 * /showusers/:
 *   get:
 *     tags:
 *       - User
 *     summary: 获取全部用户
 *     description: 获取全部已注册并激活的用户信息
 *     produces:
 *       - application/json
 *     responses:
 *       200:
 *         description: 获取成功
 *         content: application/json
 *       500:
 *         description: 获取失败
 *         content: application/json
 */
const showUsers = async ({ success, fail }) => {
    const alluser = await queryAllUser()
    return success(alluser || [], '获取成功')
}

/**
 * @swagger
 *
 * /showchats/:
 *   get:
 *     tags:
 *       - Chat
 *     summary: 获取聊天记录
 *     description: 获取群聊天记录
 *     produces:
 *       - application/json
 *     parameters:
 *       - name: current
 *         in: query
 *         required: true
 *         schema:
 *           type: integer,
 *           format: int32
 *       - name: size
 *         in: query
 *         required: true
 *         schema: 
 *           type: integer,
 *           format: int32
 *     responses:
 *       200:
 *         description: 获取成功
 *         content: application/json
 *       500:
 *         description: 获取失败
 *         content: application/json
 */
const showChats = async ({ query: { current, size }, success, fail }) => {
    const allchats = await queryAllChat({ current: Number(current || '1'), size: Number(size || '15') })
    return success(allchats || {}, '获取成功')
}

module.exports = {
    'GET /showusers': showUsers,
    'GET /showchats': showChats
}
